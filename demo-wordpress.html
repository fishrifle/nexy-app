WordPress Integration

PHP + WordPress Hooks
// functions.php
function enqueue_sponsorship_widget() {
  wp_enqueue_script(
    'sponsorship-widget',
    get_template_directory_uri() . '/js/widget.js'
  );
}
add_action('wp_enqueue_scripts', 'enqueue_sponsorship_widget');
// Use shortcode: [sponsorship_widget amount="50"]
// Or in templates: <div data-sc-widget="sponsorship"></div>
