Embeddable Sponsorship Widget

Platform: Drupal

Drupal Integration

Twig + JavaScript
# your_theme.libraries.yml
sponsorship-widget:
  js:
    js/sponsorship-widget.js: {}
# In your template:
{{ attach_library('your_theme/sponsorship-widget') }}
<div data-sc-widget="sponsorship"></div>
# Or create a custom block plugin
class SponsorshipBlock extends BlockBase {
  public function build() {
    return [
      '#markup' => '<div data-sc-widget="sponsorship"></div>',
      '#attached' => ['library' => ['your_theme/sponsorship-widget']]
    ];
  }
}
